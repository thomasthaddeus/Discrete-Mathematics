def get_data_points():
    data_point_dict = {}
    while True:
        data = input
        if data == '-1':
            break
        elif ',' != data:
            print('Error:')
        elif data.count(',') > 1:
            print('Error')
        else:
            data = data.split(',')
            try:
                data_point_dict[data[0].strip()] = int(data[1])
                print(f'Data string: {data[0]}')
                print(f'Data string: {data[1]}')
            except:
                print('Error: Comma not followed by an integer. \n')
    return data_point_dict
